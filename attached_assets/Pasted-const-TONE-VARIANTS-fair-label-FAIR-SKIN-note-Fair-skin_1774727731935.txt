function clamp(v, min, max){
return Math.max(min, Math.min(max, v));
}

function hexToRgb(hex){
const clean = hex.replace('#','').trim();
const full = clean.length === 3 ? clean.split('').map(c => c + c).join('') : clean;
const num = parseInt(full, 16);
return { r: (num >> 16) & 255, g: (num >> 8) & 255, b: num & 255 };
}

function rgbToHex(r, g, b){
return '#' + [r,g,b].map(v => clamp(Math.round(v),0,255).toString(16).padStart(2,'0')).join('');
}

function shiftColour(hex, brighten = 0, warm = 0){
const { r, g, b } = hexToRgb(hex);
const nr = clamp(r + brighten + warm, 0, 255);
const ng = clamp(g + brighten + Math.round(warm * 0.35), 0, 255);
const nb = clamp(b + brighten - Math.round(warm * 0.45), 0, 255);
return rgbToHex(nr, ng, nb);
}

function rewriteTechniqueStep(step, toneId){
let out = step;

if (toneId === 'fair') {
out = out.replace(/bronze/gi, 'rose gold').replace(/terracotta/gi, 'peach rose').replace(/copper/gi, 'champagne');
}
if (toneId === 'light') {
out = out.replace(/deep berry/gi, 'warm rose').replace(/bronze/gi, 'soft bronze');
}
if (toneId === 'medium') {
out = out.replace(/champagne/gi, 'golden champagne').replace(/peach/gi, 'apricot peach');
}
if (toneId === 'tan') {
out = out.replace(/champagne/gi, 'warm gold').replace(/pink/gi, 'coral rose').replace(/taupe/gi, 'bronzed taupe');
}
if (toneId === 'deep') {
out = out.replace(/champagne/gi, 'rich gold').replace(/soft brown/gi, 'espresso brown').replace(/pink/gi, 'berry rose').replace(/peach/gi, 'burnt peach');
}
if (toneId === 'rich') {
out = out.replace(/champagne/gi, 'molten gold').replace(/soft brown/gi, 'deep espresso').replace(/bronze/gi, 'dark bronze').replace(/pink/gi, 'mahogany rose').replace(/peach/gi, 'terracotta peach');
}

return out;
}

function buildToneSwatches(swatches, toneId, category){
const tone = TONE_VARIANTS[toneId] || TONE_VARIANTS.medium;

return swatches.map((s, idx) => {
const brighten = tone.brighten + (category === 'eye' ? -2 : 0);
const warm = tone.warm + (category === 'blush' ? 4 : 0);

const renamedPrefix =
category === 'lip' ? tone.lipPrefix :
category === 'eye' ? tone.eyePrefix :
tone.blushPrefix;

return {
n: idx === 0 ? `${renamedPrefix} ${s.n}` : s.n,
h: shiftColour(s.h, brighten, warm),
};
});
}

function tonePreferredIndex(swatches, toneId, category){
const tone = TONE_VARIANTS[toneId] || TONE_VARIANTS.medium;
const raw = category === 'lip' ? tone.lipIdx : category === 'eye' ? tone.eyeIdx : tone.blushIdx;
return Math.min(raw, swatches.length - 1);
}

function buildToneLook(baseLook, toneId){
const lip = buildToneSwatches(baseLook.lip || [], toneId, 'lip');
const eye = buildToneSwatches(baseLook.eye || [], toneId, 'eye');
const blush = buildToneSwatches(baseLook.blush || [], toneId, 'blush');
const tone = TONE_VARIANTS[toneId] || TONE_VARIANTS.medium;

return {
...baseLook,
lip,
eye,
blush,
preferred: {
lip: tonePreferredIndex(lip, toneId, 'lip'),
eye: tonePreferredIndex(eye, toneId, 'eye'),
blush: tonePreferredIndex(blush, toneId, 'blush'),
},
technique: (baseLook.technique || []).map(step => rewriteTechniqueStep(step, toneId)),
toneNote: tone.note,
toneLabel: tone.label,
};
}

function buildRecommendations(){
const recs = {};
Object.keys(BASE_LOOKS).forEach(occasionId => {
recs[occasionId] = {};
Object.keys(TONE_VARIANTS).forEach(toneId => {
recs[occasionId][toneId] = {};
['natural','glam','trendy'].forEach(lookKey => {
recs[occasionId][toneId][lookKey] = buildToneLook(BASE_LOOKS[occasionId][lookKey], toneId);
});
});
});
return recs;
}
